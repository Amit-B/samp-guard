public TimerOfTenSeconds()
{
	for(new i = 0; i < MAX_PLAYERS; i++) if(IsPlayerConnected(i) && IsPlayerUsingJoypad(i))
	{
		SendClientMessage(i,0xffffffaa,"Joypad is not allowed in this server.");
		ForceRemoveJoypad(i);
	}
	return 1;
}
